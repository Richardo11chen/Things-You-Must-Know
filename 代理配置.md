proxy_on() {
    export WINDOWS_HOST="10.0.2.2"
    export http_proxy="http://${WINDOWS_HOST}:7897"
    export https_proxy="http://${WINDOWS_HOST}:7897"
    export all_proxy="socks5://${WINDOWS_HOST}:7897"
    echo "Proxy ON: ${WINDOWS_HOST}:7897"
}

proxy_off() {
    unset http_proxy https_proxy all_proxy WINDOWS_HOST
    echo "Proxy OFF"
}

git_proxy_on() {
    export WINDOWS_HOST="10.0.2.2"
    git config --global http.proxy "http://${WINDOWS_HOST}:7897"
    git config --global https.proxy "http://${WINDOWS_HOST}:7897"
    echo "Git Proxy ON: ${WINDOWS_HOST}:7897"
}

git_proxy_off() {
    git config --global --unset http.proxy
    git config --global --unset https.proxy
    unset WINDOWS_HOST
    echo "Git Proxy OFF"
}
